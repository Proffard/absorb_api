def self.enrollments_from_collection(courses)
  enrollments = []
  api.in_parallel do
    courses.each do |course|
      enrollments << api.get("courses/#{course.id}/enrollments")
    end
  end
  enrollments.map { |response| response.body.map { |attrs| CourseEnrollment.new(attrs) } }.flatten
end
